```csharp
    var account = _dataService.GetAccounts()
        .FirstOrDefault(a => a.Name == accountName);

        _dataService.UpdateAccount(accountName, dialog.Result);
        LoadWalletData();
    }
}
```

```csharp
    .Where(t => t.Date >= thirtyDaysAgo);

MonthlyIncome = recentTransactions
    .Where(t => t.IsIncome)
    .Sum(t => t.Amount);

MonthlyExpense = recentTransactions
    .Where(t => !t.IsIncome)
    .Sum(t => t.Amount);
```

**Savings Rate:**
```csharp
SavingsRate = MonthlyIncome > 0 
    ? Math.Round((MonthlyIncome - MonthlyExpense) / MonthlyIncome * 100, 1) 
    : 0;
```
